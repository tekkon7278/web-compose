# ディレクトリ作成
mkdir -p /srv/docker/mount/web
mkdir -p /srv/docker/mount/php/public
mkdir -p /srv/docker/mount/certs
mkdir -p /srv/docker/compose

# docker compose 実行
cd /srv/docker/compose
docker compose up -d

# lets encrypt証明書取得
docker run -it --rm --name certbot -v "/srv/docker/mount/certs:/etc/letsencrypt" -v "/srv/docker/mount/web:/var/www/html/" certbot/certbot certonly -d test.tkng.site --webroot -w /var/www/html/ --agree-tos -m st6.048153@gmail.com 

# nginxのssl有効化
docker compose exec web sed -i s/#// /etc/nginx/conf.d/proxy.conf
docker compose restart web
